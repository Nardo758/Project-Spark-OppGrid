  return (
    <div className="bg-gray-50 rounded-lg p-4">

              <span className="w-5 h-5 rounded-full bg-green-100 flex items-center justify-center text-[10px]">G</span>
            </div>

            <button onClick={() => setShowResults(activeTab)} className="w-full py-3 rounded-lg text-white text-sm font-semibold" style={{ background: "#1C1917" }}>
              ✦ Analyze
            </button>
          </div>

          {/* RIGHT: Report Selection + Bundles + Stripe */}
          <div className="col-span-2 p-5">
            <label className="block text-xs font-medium text-gray-700 mb-2">Select report type</label>

            {/* Dropdown */}
            <select
              value={selectedReport?.slug || ""}
              onChange={(e) => {
                if (!e.target.value) { setSelectedReport(null); return }
                const all = REPORT_CATEGORIES.flatMap(c => c.reports)
                const found = all.find(r => r.slug === e.target.value)
                if (found) setSelectedReport(found)
              }}
              className="w-full p-2.5 border border-gray-200 rounded-lg text-sm bg-white mb-2"
            >
              <option value="">Choose a report...</option>
              {REPORT_CATEGORIES.map(cat => (
                <optgroup key={cat.name} label={`${cat.icon} ${cat.name}`}>
                  {cat.reports.map(r => (
                    <option key={r.slug} value={r.slug}>{r.name} — {r.price}</option>
                  ))}
                </optgroup>
              ))}
            </select>

            {/* Bundle Toggle */}
            <button
              onClick={() => setShowBundles(!showBundles)}
              className="flex items-center gap-1 text-xs font-medium mb-2 hover:underline"
              style={{ color: "#0F6E56" }}
            >
              💎 {showBundles ? "Hide bundles" : "View bundle deals (save up to 83%)"}
            </button>

            {/* Bundles */}
            {showBundles && (
              <div className="space-y-1.5 mb-3">
                {[
                  { name: "Starter Bundle", badge: "MOST POPULAR", price: "$329", desc: "Validation + Pitch package for fundraising", saves: "Save $53 (14%)", count: "4 reports included", featured: true },
                  { name: "Strategic Analysis Bundle", price: "$229", desc: "Complete competitive and environmental intelligence", saves: "Save $58 (20%)", count: "3 reports included" },
                  { name: "Professional Bundle", price: "$549", desc: "Complete execution package - replaces $30,000+ in consulting", saves: "Save $120 (18%)", count: "7 reports included" },
                  { name: "Consultant License", price: "$2499", desc: "Unlimited reports for 25 opportunities per year", saves: "Save $12500 (83%)", count: "7 reports included" },
                ].map(b => (
                  <button
                    key={b.name}
                    onClick={() => setSelectedReport({ slug: b.name.toLowerCase().replace(/ /g, "_"), name: b.name, price: b.price, bundle: true })}
                    className={`w-full text-left p-2.5 rounded-lg border-2 transition-colors ${b.featured ? "border-yellow-400 bg-yellow-50/30" : "border-gray-200 hover:border-gray-300"}`}
                  >
                    <div className="flex items-start justify-between">
                      <div className="flex-1 min-w-0">
                        <div className="flex items-center gap-1.5 flex-wrap">
                          <span className="text-xs font-semibold text-gray-900">{b.name}</span>
                          {b.badge && (
                            <span className="px-1.5 py-0.5 rounded text-[8px] font-bold text-white" style={{ background: "#D97757" }}>{b.badge}</span>
                          )}
                        </div>
                        <p className="text-[10px] text-gray-500 mt-0.5 leading-tight">{b.desc}</p>
                        <p className="text-[10px] font-medium mt-0.5" style={{ color: "#0F6E56" }}>{b.saves}</p>
                      </div>
                      <div className="text-right ml-2 flex-shrink-0">
                        <div className="text-base font-bold text-gray-900">{b.price}</div>
                        <div className="text-[9px] text-gray-400">{b.count}</div>
                      </div>
                    </div>
                  </button>
                ))}
              </div>
            )}

            {/* Selected Report Preview (when individual report chosen) */}
            {selectedReport && !selectedReport.bundle && <ReportPreview report={selectedReport} />}

            {/* Generate Button */}
            <button
              onClick={() => { if (showResults && selectedReport) setView("report") }}
              disabled={!showResults || !selectedReport}
              className={`w-full py-3 rounded-lg text-sm font-semibold transition-colors flex items-center justify-center gap-2 ${showResults && selectedReport ? "text-white cursor-pointer shadow-sm" : "bg-gray-200 text-gray-400 cursor-not-allowed"}`}
              style={showResults && selectedReport ? { background: "linear-gradient(135deg, #7C3AED, #6D28D9)" } : {}}
            >
              <span>✦</span>
              {!showResults && !selectedReport ? "Run analysis first ↑" :
               showResults && !selectedReport ? "Select a report →" :
               !showResults && selectedReport ? "Run analysis first ↑" :
               `Generate Report`}
            </button>

            {/* Stripe Trust Signals */}
            <div className="flex flex-col items-center gap-1 mt-3">
              <div className="flex items-center gap-3 text-[11px] text-gray-400">
                <span>○ Secure payment</span>
                <span>☑ Money-back guarantee</span>
              </div>
              <div className="flex items-center gap-1.5 text-[11px] text-gray-400">
                <span className="font-bold tracking-tight" style={{ color: "#635BFF" }}>stripe</span>
                <span>Powered by Stripe</span>
              </div>
            </div>
          </div>
        </div>
      </div>

      {/* Results below the unified card */}
      {showResults === "validate" && <ValidateResults />}
      {showResults === "search" && <SearchResults />}
      {showResults === "location" && <LocationResults />}
      {showResults === "clone" && <CloneResults />}

      {!showResults && (
        <div className="text-center py-12 text-gray-400 text-sm">
          Run an analysis to see enriched results, then select a report to generate
        </div>
      )}
    </div>
  )
}
