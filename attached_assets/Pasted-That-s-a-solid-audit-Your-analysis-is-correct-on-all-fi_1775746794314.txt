async def generate_free_report(..., db):
    async with db.transaction():
        # Lock the row and check+increment atomically
        row = await db.fetch_one("""
            SELECT free_reports_used, free_reports_limit 
            FROM user_usage 
            WHERE user_id = :user_id
            FOR UPDATE
        """, {"user_id": user_id})
        
        if row["free_reports_used"] >= row["free_reports_limit"]:
            raise HTTPException(403, "No free reports remaining")
        
        # Increment BEFORE generation
        await db.execute("""
            UPDATE user_usage 
            SET free_reports_used = free_reports_used + 1
            WHERE user_id = :user_id
        """, {"user_id": user_id})
    
    # Now generate (outside transaction — don't hold lock during AI call)
    try:
        report = await create_report(...)
        result = await run_ai_generation(report)
        return result
    except Exception as e:
        # Rollback the credit on failure
        await db.execute("""
            UPDATE user_usage 
            SET free_reports_used = free_reports_used - 1
            WHERE user_id = :user_id
        """, {"user_id": user_id})
        raise

async def _handle_slot_purchase(session: dict, db):
    session_id = session["id"]
    user_id = session["metadata"]["user_id"]
    slots = int(session["metadata"]["slots"])
    
    # Check if already fulfilled
    existing = await db.fetch_one("""
        SELECT id FROM slot_purchases 
        WHERE stripe_session_id = :session_id
    """, {"session_id": session_id})
    
    if existing:
        logger.info(f"Slot purchase {session_id} already fulfilled, skipping")
        return
    
    # Record fulfillment BEFORE adding slots
    await db.execute("""
        INSERT INTO slot_purchases (stripe_session_id, user_id, slots, created_at)
        VALUES (:session_id, :user_id, :slots, NOW())
    """, {"session_id": session_id, "user_id": user_id, "slots": slots})
    
    await slot_service.add_bonus_slots(user_id, slots)
